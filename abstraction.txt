class Op3FallControlEnv(gymnaisum.Env)

    # Observation space configurations (these add components to observation space)
    # values should be normalized
    def obs_joint(List joint_names, bool angle=True, bool velocity=True)
    def obs_fall_vel() -> continuous fall_speed = sqrt(wx² + wy²)
    def obs_fall_vel_discrete() -> {NORMAL/FAST}
    def obs_fall_angle() -> radians, fall_angle = atan2(wx, wy)
    def obs_fall_angle_discrete(int dirs=FallAngle.FOUR) -> {F/B/L/R/}, else if dirs==FallAngle.EIGHT {F/B/L/R/FL/FR/BL/BR}

    # Action space configurations (these add components to action space)
    # action output values should be normalized
    def act_joint_pos(List joint_names) -> outputs angle_delta (target_q = current_q + angle_delta)
    def act_joint_delta(List joint_names) -> outputs target_q
    
    # Core function
    _get_obs()
    _get_info()
    _apply_action()
    reset()    
    step()
    get_obs_dim()
    get_act_dim()
    def close():
        if self.window is not None:
            pygame.display.quit()
            pygame.quit()
    
    _render_frame() -> returns a single frame representing the current state of the environment. A frame is a np.ndarray with shape (x, y, 3) representing RGB values for an x-by-y pixel image.
    render() -> returns _render_fram() if self.render_mode == "rgb_array"

    # Abstract method, must be implemented by subclass
    def compute_reward()
